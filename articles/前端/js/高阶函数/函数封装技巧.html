# 日期格式化

通过封装一个日期格式化函数来理解参数归一化

我们需要封装一个函数，根据不同的参数进行不同的格式化，如果有默认的参数无法满足需求的情况，甚至可以传入一个回调函数作为参数，自定义格式化需求

function formatDate(date, formatter, isPad = false) {
    // 获取参数归一化后的格式化工具函数，这个时候formatter这个参数已经被处理成了一个函数
	const formatter = _normalize(formatter);
	const dateInfo = {
		yyyy: date.getFullYear().toString(),
		MM: (date.getMonth() + 1).toString(),
		dd: date.getDate().toString(),
		HH: date.getHours().toString(),
		mm: date.getMinutes().toString(),
		ss: date.getSeconds().toString(),
		ms: date.getMilliseconds().toString()
	}
	// 工具函数，用于补零
	function _pad(prop, len) {
		dateInfo[prop] = dateInfo[prop].padStart(len, '0');
	}
	// 补零
	if(isPad) {
		_pad('yyyy', 4);
		_pad('MM', 2);
		_pad('dd', 2);
		_pad('HH', 2);
		_pad('mm', 2);
		_pad('ss', 2);
		_pad('ms', 2);
	}
	// 使用格式化函数对日期进行格式化
	const result = formatter(dataInfo);
	return result;
}

// 不允许直接调用的工具函数，用于参数归一化，将不同的参数都转化为函数
function _normalize(formatter) {
	if(typeof formatter === 'function') {
		return formatter
	}
	if(typeof formatter !== 'string') {
		throw new Error('formatter is must a string');
	}
	if(typeof formatter === 'date') {
		formatter = 'yyyy-MM-dd';
	} else if(typeof formatter === 'datetime') {
		formatter = 'yyyy-MM-dd HH:mm:ss';
	}

	return (dateInfo) => {
		const {yyyy, MM, dd, HH, mm, ss, ms} = dateInfo;
		return formatter
			.replaceAll('yyyy', yyyy)
			.replaceAll('MM', mm)
			.replaceAll('dd', dd)
			.replaceAll('HH', HH)
			.replaceAll('mm', mm)
			.replaceAll('ss', ss)
			.replaceAll('ms', ms)
	}
}

这样封装之后这个函数就可以使用多种不同的方式进行调用，如果不使用参数归一化，这个函数将会变得十分臃肿。

// 这些调用方式都是可以的
formatDate(new Date(), 'date');
formatDate(new Date(), 'date', true);
formatDate(new Date(), 'datetime');
formatData(new Date(), 'yyyy-MM-dd HH:mm:ss');
formatData(new Date(), (dateInfo) => {}, true);

# 数据分组

下面再写一个数据分组函数，给定一个数组，数组中有若干个对象，希望按照某一个属性的值对数据进行分组，很容易写出下面的函数

function groupBy(list, prop) {
	const result = {};
	for(const item of list) {
		const key = item[prop];
		if(!result[key]) {
			result[key] = []
		}
		result[key].push(item);
	}
	return result;
}

但如果我有更复杂的分组要求，比如我需要同时使用两个属性来分组。那就需要交给用户一个函数，来让用户自定义分组方式

function groupBy(list, getKey) {
	const result = {};
	for(const item of list) {
		// 将参数由属性名，改为获取属性名的函数
		const key = getKey(item);
		if(!result[key]) {
			result[key] = []
		}
		result[key].push(item);
	}
	return result;
}

// 调用示例
groupBy(list, (item) => `${item.name}-${item.age}`)
// 但是对于之前只需要一个属性名来分类的情况这样调用就显得难受
groupBy(list, (item) => item.name); // 能否只传一个属性名

对于只需要一个属性作为分组依据的情况，传入函数显得难受，这就又用到了参数归一化

function groupBy(list, getKey) {
	const result = {};
	// 如果传入的是字符串则当作属性名处理
	if(typeof getKey === 'string') {
		const prop = getKey;
		getKey = (item) => item[prop];
	}
	
	for(const item of list) {
		// 将参数由属性名，改为获取属性名的函数
		const key = getKey(item);
		if(!result[key]) {
			result[key] = []
		}
		result[key].push(item);
	}
	return result;
}

// 这样依赖下面两种调用方式都可以了
groupBy(list, (item) => `${item.name}-${item.age}`)
groupBy(list, 'age');

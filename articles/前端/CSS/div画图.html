# 铅笔

div {
      width: 120px;
      height: 240px;
      background: 
        linear-gradient(#e53466 70%,
        #d8d8d8 70%) no-repeat,
        linear-gradient(to right,
        #f06a35 33.3%,
        #f68c30 33.3%,
        #f68c30 66.6%,
        #ffa927 66.6%);
      background-size: 100% 30%, 100% 70%;
      border-radius: 30px 30px 0 0;
      position: relative;
      transform: rotate(30deg);
    }

    div::before {
      content: '';
      width: 120px;
      height: 20px;
      position: absolute;
      bottom: 0;
      /* 径向渐变，以20，20为圆心 */
      background: radial-gradient(circle at 20px 20px, #f4c09b 20px, transparent 0);
      background-size: 40px 20px;
    }
    div::after {
      content: "";
      width: 120px;
      height: 110px;
      /* 线性渐变，第二个参数为渐变范围 */
      background: linear-gradient(#f4c09b 70%, #5e381f 70%);
      position: absolute;
      bottom: -110px;
      /* 裁剪属性，类似于蒙版 */
      /* polygon()函数创建一个三角形的裁剪，指定三个顶点来绘制三角形
         还有其他的形状函数例如circle()、ellipse()、inset() */
      clip-path: polygon(0 0, 100% 0, 50% 100%);
    }

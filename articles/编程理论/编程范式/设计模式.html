# 工厂模式

根据不同的需要，返回同一类的不同的对象

class User {
  static UserFactory(role) {
    switch(role) {
      case 'superadmin': 
        return new User(role, ['home', 'super', 'user', 'message']);
        break;
      case 'admin':
        return new User(role, ['home', 'user', 'message']);
        break;
      case 'editor':
        return new User(role, ['home', 'message']);
        break;
      default: 
        throw new Error("参数错误");
    }
  }
  
  constructor(role, pages) {
    this.role = role;
    this.pages = pages;
  }
}

# 单例模式

// es5 闭包实现
const Singleton = (function() {
  let instance;
  
  function User(name, age) {
    this.name = name;
    this.age = age;
  }
  
  return function(name, age) {
    if(!instance) {
      instance = new User(name, age);
    }
    return instance;
  }
})

// es6 类实现
class Singleton {
  constructor(name, age) {
    if(!Singleton.instance) {
      this.name = name;
      this.age = age;
      Singleton.instance = this;
    }
    
    return Singleton.instance;
  }
}

# 装饰器模式

装饰器模式可以很方便得提高扩展性，axios的拦截器、react高阶组件就是典型的装饰器模式

Function.prototype.before = function(beforeFn) {
  const _this = this;
  return function() {
    before.apply(this, arguments);
    return _this.apply(this, arguments);
  }
}

Function.prototype.after = function(beforeFn) {
  const _this = this;
  return function() {
    const result = _this.apply(this, arguments);
    before.apply(this, arguments);
    return result;
  }
}

# 策略模式

定义一系列算法，并将其封装起来，使得算法可以自由替换，并且避免大量的if else，常用的根据状态码来显示不同的内容就可以使用策略模式来简化代码

const strategry = {
  "A": (salary) => {
    return salary * 4;
  },
  "B": (salary) => {
    return salary * 3;
  },
  "C": (salary) => {
    return salary * 2;
  },
}

function calBonus(level, salary) {
  return strategry[level](salary);
}

# 代理模式

为一个对象提供一系列操作方法来限制对他的访问，vue就是通过代理模式实现的响应式

const obj = {
  name: 'neko',
  price: 10000
};

const proxy = new Proxy(obj, {
  get(target, key) {
    if(key == price) {
      console.log("访问");
    }
    
    return target[key];
  },
  
  set(target, key, value) {
  	if(key === price) {
      console.log("设置了");
      if(value > 10000) {
        console.log("成交");
        target[key] = value;
      } else {
        throw new Error("加钱！");
      }
    }
  }
})

# 观察者模式

当一个对象更新后，所有依赖他的对象都被自动更新，vue中的watch就是观察者模式

// 被观察者更新后
class Subject {
  constructor() {
    this.observers = [];
  }
  
  add(observer) {
    this.observers.push(observer);
  }
  
  remove(observer) {
    this.observers  = this.observers.filter(item => item !== observer);
  }
  notify() {
    this.observers.forEach(item => {
      item.updata();
    })
  }
}

  }
